```JavaScript
// class 实现继承
class Animal {
    constructor () {
        this.body = '身体'
        this._race = '运物'
    }
    move () {
        console.log('我能动')
    }
    get race () {
        return this._race
    }
    set race (value) {
        this._race = value
        // 控制写的例子
        if (value.length > 4) {
            return '滚'
        } else {
            this._race = value
        }
    }
}
// 人类从动物这个类中继承(拓展)
class Person extends Animal {
    constructor (name) {
        super() // 执行所继承类的constructor
        this.name = name
        this.age = 18
        this.body = this.body + '手脚'
    }
    // 静态方法
    static die () {
        console.log('die')
    }
    walk () {
        console.log('man can walk')
    }
}

var p1 = new Person('sam')
```
