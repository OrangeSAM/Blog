```
var commonProperty = {
    hi (a) {
        console.log(a)
    }
}
var obj = {name: 'sam'}
obj.__proto__ = 公用属性
obj.hi()
// undefined
这样并不能打印出 name sam
因为当执行obj.hi()时, 找到了原型上的hi是这样执行的.
obj.__proto__.hi.call(obj)
obj 在hi 方法中并不是第一个参数, 而是隐藏在this中.


var a = {
    hi (b) {
        console.log(this, b, arguments)
    }
}
var obj = {name: 'sam'}

obj.__proto__ = a
obj.hi(1)
// 这是 b 才是1, argumens是一个隐藏参数数组,
```
